\begin{enumerate}
\item Express the distance $s$ in terms of the height $h$ of the tree above eye level (measured in feet) and the angles $\theta_1$, $\theta_2$.

\item Express the height of the tree above eye level in terms of $s$ and the angles $\theta_1$, $\theta_2$.

\end{enumerate}
\end{question}
